00502 
00503           // For each statement in the body, get the right and left hand sides
00504           // of each field assignment.

00517                Set<Expression> rightHandSideExpressions = assignmentVisitor
00518                          .getRightHandSideAssignmentExpressions();
00519 
00520                // For each expression that assigns a field, add it to the map
00521                for (Expression expression : leftHandSideExpresions) {
00522 
00523                     // Get the field accessed in the left hand side (should only be
00524                     // one)
00525                     expression.accept(fieldVisitor);
00526                     Set<IVariableBinding> fields = fieldVisitor.getFields();
00527 
00528                     // Add its received value (expression) to the map
00529                     for (IVariableBinding iVariableBinding : fields) {
00530                          for (Expression rhsExpression : rightHandSideExpressions) {
00531                               finalFieldAssignments.put(iVariableBinding, rhsExpression);
00532                          }
