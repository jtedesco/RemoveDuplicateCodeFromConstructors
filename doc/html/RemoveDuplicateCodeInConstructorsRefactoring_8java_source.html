00972 
00973           for (MethodDeclaration constructor : parameterFieldsInConstructorAssignments.keySet()) {

00976                // long as this constructor is not that one, it should delegate its

00979                          || !constructor.equals(constructorAssigningAllAssignedFields)) {

01004      private MethodDeclaration getConstructorContainingAllAssignedFields(RefactoringStatus status) {
01005 
01006           // Get the alphabetically ordered list of fields we're looking for
01007           List<IVariableBinding> fieldsAssignedInAnyConstructors = new ArrayList<IVariableBinding>();
01008           buildFieldsAssignedInOrder(fieldsAssignedInAnyConstructors);
01009 
01010           String expectedMasterConstructorSignature = null;
01011 
01012           // Find the set of fields assigned in any constructor
01013           for (MethodDeclaration constructor : constructorDeclarations) {
01014 
01015                // Get the fields assigned in this constructor
01016                Set<IVariableBinding> fieldAssignedFromParametersInThisConstructor = parameterFieldsInConstructorAssignments
01017                          .get(constructor);
01018 
01019                // Build the signature for this constructor
01020                List<SingleVariableDeclaration> parameters = constructor.parameters();
01021                String constructorSignature = constructor.getName().toString() + " ";
01022                for (SingleVariableDeclaration parameter : parameters) {
01023                     constructorSignature += parameter.getType().toString() + " ";
01024                }
01025 
01026                // Build the the expected signature of the master constructor (if we
01027                // haven't yet)
01028                if (expectedMasterConstructorSignature == null) {
01029                     expectedMasterConstructorSignature = constructor.getName().toString() + " ";
01030                     for (IVariableBinding field : fieldsAssignedInAnyConstructors) {
01031                          expectedMasterConstructorSignature += field.getType().getName() + " ";
01032                     }
01033                }
01034 
01035                // If this constructor assigns its parameters to all fields used in
01036                // the class, this is our constructor
01037                if (fieldAssignedFromParametersInThisConstructor.equals(new HashSet(fieldsAssignedInAnyConstructors))) {
01038 
01039                     return constructor;
01040 
01041                     // If we found a constructor with the same parameters as our
01042                     // master constructor is going to need, fail
01043                } else if (expectedMasterConstructorSignature.equals(constructorSignature)) {
01044 
01045                     RefactoringStatus errorStatus = RefactoringStatus
01046                               .createFatalErrorStatus(RemoveDuplicateCodeInConstructorsConstants.UNABLE_TO_REFACTOR);
01047                     status.merge(errorStatus);
01048                     return null;
01049 
01050                }
01051           }
01052           return null;
